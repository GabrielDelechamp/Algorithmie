Var saisie, resultat, index1, index2, SommeIndex : Entier
Var Tab : Tableau
Début
    Tab = []
    resultat=0
    index1=0
    index2=0
    i=0

    Afficher('Entrer un nombre : ')
    Récupérer(saisie)

    Tant que saisie != 0 :
        Afficher('Entrer un nombre : ')
        Récupérer(saisie)
        VerifEntier(saisie)
        resultat+=saisie
        Tab[i]=saisie
        i+=1
    Fin Tant que

    Si resultat >=100:
        Sortir de la boucle

    index1=SaisieIndex(index1)

    index2=SaisieIndex(index2)
    
    SommeIndex=Tab[index1]+Tab[index2]

    Afficher(SommeIndex)

Fin

FONCTION VerifEntier (saisieF:Entier):Entier

    Tant que saisieF is not Entier
        Afficher('Entrer un Entier : ')
        Récupérer(saisieF)
    Fin Tant que

    Retourne(saisieF):Entier
Fin

FONCTION SaisieIndex (index:Entier):Entier

    Afficher('Saisir un index : ')
    Récupérer(index)
    VerifEntier(index)

    Tant que Longueur(Tab) < index :
        Afficher('Saisir un index valide : ')
        Récupérer(index)
        VerifEntier(index)
    Fin Tant que

    Retourne index
Fin
